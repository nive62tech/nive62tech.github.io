from fastapi import FastAPI, UploadFile
from ollama import Client
import pandas as pd

app = FastAPI(title="Vizora API")
llm = Client(host="http://localhost:11434")

@app.post("/analyse")
async def analyse_data(file: UploadFile):
    """Parse CSV/Excel, query local LLM — zero cloud, full privacy."""
    df = pd.read_csv(file.file)
    schema = df.dtypes.to_dict()

    response = llm.chat(
        model="mistral",
        messages=[{
            "role": "user",
            "content": f"Given schema {schema}, suggest chart types as JSON"
        }]
    )
    return {"suggestions": response["message"]["content"]}
from fastapi.middleware.cors import CORSMiddleware

def register_middleware(app):
    app.add_middleware(
        CORSMiddleware,
        allow_origins=["http://localhost:3000"],
        allow_methods=["*"],
        allow_headers=["*"],
    )

@app.get("/health")
def health():
    return {"status": "ok", "model": "mistral"}
import { useState, useEffect } from "react";
import { LineChart, Line, XAxis, YAxis, Tooltip } from "recharts";

export default function DynamicChart({ data, suggestion }) {
  const [chartType, setChartType] = useState("line");

  useEffect(() => {
    if (suggestion?.type) setChartType(suggestion.type);
  }, [suggestion]);

  return (
    <div className="chart-container">
      <LineChart width={600} height={300} data={data}>
        <XAxis dataKey="name" />
        <YAxis />
        <Tooltip />
        <Line type="monotone" dataKey="value" stroke="#00c2ff" />
      </LineChart>
    </div>
  );
}
from dataclasses import dataclass
from typing import Literal

@dataclass
class ChartSuggestion:
    chart_type: Literal["line","bar","scatter","pie"]
    x_col: str
    y_col: str
    title: str
    color: str = "#00c2ff"

def parse_llm_response(raw: str) -> ChartSuggestion:
    import json
    data = json.loads(raw)
    return ChartSuggestion(**data)
